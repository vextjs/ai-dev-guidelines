## 一、鉴权引擎（src/check/）

### 1.1 checker.ts — 主流程

**构造器**：`new Checker(storage, cache, strict)`

```typescript
import type { StorageAdapter } from '../storage/adapter';
import type { PermissionCache } from '../cache/permission-cache';
import type { PermissionRule } from '../types';
import { matchRule, matchAction } from './wildcard';
import { Resolver } from './resolver';

class Checker {
  private storage: StorageAdapter;
  private cache: PermissionCache;
  private strict: boolean;
  private resolver: Resolver;

  constructor(storage: StorageAdapter, cache: PermissionCache, strict: boolean) {
    this.storage = storage;
    this.cache = cache;
    this.strict = strict;
    this.resolver = new Resolver();
  }
}
```

**`can` 核心流程**（含 🆕 write 双向语义修复）：

```
async can(userId: string, action: string, resource: string): Promise<boolean>

  // 🆕 v2.3 write 双向语义 — 请求中的 write 展开为 create AND update
  if action === 'write':
    const canCreate = await this._canSingle(userId, 'create', resource)
    const canUpdate = await this._canSingle(userId, 'update', resource)
    return canCreate && canUpdate

  return this._canSingle(userId, action, resource)


async _canSingle(userId: string, action: string, resource: string): Promise<boolean>

  rules = await _getRules(userId)
  // userId=null/undefined → getUserRoles([]) → rules=[] → return false（匿名用户）

  if strict:                                   // 默认 true
    for rule of rules where type==='deny':
      if matchRule(rule, action, resource): return false   // deny 命中立即拒绝
    for rule of rules where type==='allow':
      if matchRule(rule, action, resource): return true    // allow 命中通过
    return false                               // 无 allow 命中 → 默认拒绝（白名单）

  else (strict=false):
    result = false
    for rule of rules:
      if matchRule(rule, action, resource):
        result = (rule.type === 'allow')       // 最后命中的规则胜出
    return result


async _getRules(userId: string): Promise<PermissionRule[]>
  cached = cache.get(userId)
  if cached: return cached
  roleIds = await storage.getUserRoles(userId)
  rules = await resolver.mergeRules(roleIds, storage, strict)
  cache.set(userId, rules)
  return rules
```

> 🆕 **v2.3 write 双向语义**：当请求 action 为 `'write'` 时，`checker.can()` 在顶层展开为 `can('create') && can('update')`（AND 语义）。这确保了以下两种配置方式语义一致：
>
> - 配置 `allow('editor', 'write', 'db:articles')` → `can(userId, 'create', ...)` ✅、`can(userId, 'write', ...)` ✅
> - 配置 `allow('editor', 'create', ...)` + `allow('editor', 'update', ...)` → `can(userId, 'write', ...)` ✅（两者都有才返回 true）
>
> 详见 [API设计 §八 write 双向语义](./03-API设计.md)

**`filterFields` 流程**：

```
async filterFields(
  userId: string,
  action: string,
  resource: string,
  data: Record<string, unknown>
): Promise<Partial<Record<string, unknown>>>

  ⓪ this._checkInit()                 // 🔴 初始化保护（未 init 抛 NOT_INITIALIZED）
  ① 校验 resource 以 'db:' 开头，否则抛 INVALID_RESOURCE_PATH
  ② 校验 data 为非 null 普通对象，否则抛 INVALID_ARGUMENT
  ③ clone = Object.assign({}, data)    // 浅克隆（v1.0.0 只处理顶层字段）
  ④ for key of Object.keys(clone):
       fieldResource = resource + ':' + key   // 'db:users:email'
       if !await can(userId, action, fieldResource): delete clone[key]
  ⑤ return clone
```

> ⚠️ **v2.3 签名变更**：`action` 参数从最后一个可选参数调整为第二个必填参数，与 `can(userId, action, resource)` 保持一致。写入过滤场景需要明确传 `'update'` 或 `'create'`。

**`getResources` 流程**（deny 过滤使用通配匹配）：

```
async getResources(userId: string, action: string = 'invoke'): Promise<string[]>

  rules = await _getRules(userId)
  allows = rules.filter(r => r.type==='allow' && matchAction(r.action, action))
  denies = rules.filter(r => r.type==='deny'  && matchAction(r.action, action))
  result: string[] = []
  for allowRule of allows:
    // strict=true: 检查 deny pattern 是否覆盖了 allow resource
      // 🔴 参数方向: matchResource(denyPattern, allowResource)
      //    含义: "deny 规则的资源路径（作为 pattern）能覆盖 allow 规则的资源路径吗？"
      //    当 deny 为通配（如 *:/api/users/*）、allow 为精确路径时 → 正确过滤
      //    当 deny 为精确路径、allow 为通配时 → 不会过滤（假阳性，已在文档中标注）
      covered = strict && denies.some(d => matchResource(d.resource, allowRule.resource))
      if !covered: result.push(allowRule.resource)
  return [...new Set(result)]
  // action='invoke' → 只含接口资源（db: 规则 action 是 read 等，matchAction 不命中）
  // action='read'   → 只含 db: 资源
  // action='*'      → 混合返回（不推荐）
```

> ⚠️ **假阳性限制**：当 deny 为精确规则（如 `DELETE:/api/users/123`）而 allow 为通配符（如 `*:/api/users/*`）时，`matchResource(deny.resource, allow.resource)` 返回 false，allow 不会被过滤。这意味着 `getResources` 返回的通配符资源可能包含实际被 deny 的具体路径。前端最终权限判断应以 `can()` 为准，`getResources` 仅用于粗粒度菜单渲染。

### 1.2 resolver.ts — 继承链解析

```typescript
import type { StorageAdapter } from '../storage/adapter';
import type { PermissionRule } from '../types';
import { PermiaError, PermiaErrorCode } from '../core/errors';

class Resolver {
  // BFS 展开单个角色继承链
  async resolveRoleChain(roleId: string, storage: StorageAdapter): Promise<string[]> {
    const visited = new Set<string>();
    const queue: string[] = [roleId];
    const result: string[] = [];

    while (queue.length > 0) {
      const current = queue.shift()!;
      if (visited.has(current)) {
        throw new PermiaError(PermiaErrorCode.CIRCULAR_INHERITANCE, `Circular inheritance detected at role '${current}'`);
      }
      visited.add(current);
      result.push(current);

      const role = await storage.getRole(current);
      if (!role) continue;   // 容错：已删除角色直接跳过（orphan roleId）
      if (role.parent) {
        if (visited.has(role.parent)) {
          throw new PermiaError(PermiaErrorCode.CIRCULAR_INHERITANCE, `Circular inheritance: '${current}' → '${role.parent}'`);
        }
        queue.push(role.parent);
      }
    }
    return result;   // [self, parent, grandparent, ...]  子→父顺序
  }

  // 合并多个角色的规则
  async mergeRules(roleIds: string[], storage: StorageAdapter, strict: boolean): Promise<PermissionRule[]> {
    const allRules: PermissionRule[] = [];
    const processedIds = new Set<string>();

    for (const roleId of roleIds) {
      const chain = await this.resolveRoleChain(roleId, storage);
      for (const id of chain) {
        if (processedIds.has(id)) continue;  // 去重：多角色共同继承同一父角色时，规则只追加一次
        processedIds.add(id);
        const rules = await storage.getRules(id);
        allRules.push(...rules);
      }
    }

    if (strict) {
      // deny 前置，确保 can() 遍历 deny 时在 allow 之前
      return [
        ...allRules.filter(r => r.type === 'deny'),
        ...allRules.filter(r => r.type === 'allow'),
      ];
    }
    return allRules;
  }
}
```

### 1.3 wildcard.ts — 路径匹配（同时对外导出）

```typescript
import type { PermissionRule } from '../types';

// 模块导出
export { matchResource, matchAction, matchRule };

/**
 * 匹配资源路径
 *
 * 签名语义: matchResource(pattern, resource)
 *   - pattern  = 规则中定义的资源路径（可含通配符 *）
 *   - resource = 实际请求的资源路径（通常是精确路径）
 *   - 返回 true 表示 pattern 覆盖了 resource
 */
function matchResource(pattern: string, resource: string): boolean {
  if (pattern === '*') return true;               // 全局通配
  if (pattern === resource) return true;           // 精确匹配（快速路径）

  // ① 先判断 db 资源（明确前缀，无歧义）
  if (pattern.startsWith('db:') || resource.startsWith('db:')) {
    // 两者都必须是 db 资源才能匹配
    if (!pattern.startsWith('db:') || !resource.startsWith('db:')) return false;

    const pParts = pattern.split(':');
    const rParts = resource.split(':');
    // 🔴 段数必须一致（db:* 只匹配 db:<collection>，不跨段匹配 db:<collection>:<field>）
    if (pParts.length !== rParts.length) return false;
    return pParts.every((p, i) => p === '*' ? true : p === rParts[i]);
  }

  // ② 接口资源 fallback（db 分支已 return，剩余均视为接口资源）
  // 覆盖: 'GET:/api/users'、'*:/api/*'、'*:*' 等所有含 ':' 的非 db 资源
  // 🔴 v2.5 修复: 移除原 `includes(':/') ` 条件判断 — '*:*' 不含 ':/' 但语义为"所有接口"，
  //    旧条件导致 matchResource('*:*', 'GET:/api/users') 错误返回 false
  const [pMethod, pPath] = splitFirst(pattern, ':');
  const [rMethod, rPath] = splitFirst(resource, ':');

  // pattern 或 resource 中至少一个需要含 ':'，否则无法拆分为 method:path
  if (!pMethod || !rMethod) return false;

  if (pMethod !== '*' && pMethod !== rMethod) return false;
  return pathMatch(pPath, rPath);
}

/** 路径段匹配 */
function pathMatch(patternPath: string, resourcePath: string): boolean {
  const pSegs = patternPath.split('/').filter(Boolean);
  const rSegs = resourcePath.split('/').filter(Boolean);

  for (let i = 0; i < pSegs.length; i++) {
    if (pSegs[i] === '*' && i === pSegs.length - 1) {
      // 末段 * → 匹配剩余所有段（至少 1 段）
      return rSegs.length > i;
    }
    if (i >= rSegs.length) return false;
    if (pSegs[i].startsWith(':')) continue;   // 路径参数匹配任意单段
    if (pSegs[i] !== rSegs[i]) return false;  // 精确匹配（大小写敏感）
  }
  return pSegs.length === rSegs.length;
}

/**
 * action 匹配（规则中的 action 作为 pattern）
 *
 * ⚠️ 此函数只处理「规则 → 请求」方向的匹配。
 * 「请求中 write」的双向展开在 checker.can() 层面处理。
 */
function matchAction(pattern: string, action: string): boolean {
  if (pattern === '*') return true;
  if (pattern === action) return true;
  // write = create + update（规则侧展开，OR 语义）
  if (pattern === 'write' && (action === 'create' || action === 'update')) return true;
  return false;
}

/** 规则匹配（组合） */
function matchRule(rule: PermissionRule, action: string, resource: string): boolean {
  return matchAction(rule.action, action) && matchResource(rule.resource, resource);
}

/** 辅助：按第一个分隔符分割（仅分割一次） */
function splitFirst(str: string, sep: string): [string, string] {
  const idx = str.indexOf(sep);
  if (idx === -1) return [str, ''];
  return [str.slice(0, idx), str.slice(idx + 1)];
}
```

### 2.2 FileAdapter（默认适配器）

> ⚠️ **生产环境注意**：容器化部署需挂载数据卷保证持久化；Serverless 环境（Lambda / Cloud Functions）无持久化文件系统，FileAdapter 不适用；多进程部署（PM2 cluster）各进程独立内存副本存在数据竞争，需改用 MonSQLizeAdapter。

```typescript
import { StorageAdapter } from './adapter';
import type { RoleData, PermissionRule } from '../types';
import { PermiaError, PermiaErrorCode } from '../core/errors';
import * as fs from 'node:fs/promises';

interface FileAdapterOptions {
  path?: string;    // 默认 './permia-data.json'
}

interface FileData {
  roles:     Record<string, RoleData>;
  userRoles: Record<string, string[]>;
  rules:     Record<string, PermissionRule[]>;
}

class FileAdapter extends StorageAdapter {
  private filePath: string;
  private _data: FileData = { roles: {}, userRoles: {}, rules: {} };
  private _roleUsers: Map<string, Set<string>> = new Map();   // 反向索引
  private _debounceTimer: ReturnType<typeof setTimeout> | null = null;

  constructor(options?: FileAdapterOptions) {
    super();
    this.filePath = options?.path ?? './permia-data.json';
  }
}
```

**生命周期**：

```
init():
  → fs.readFile(path, 'utf-8') 异步读
  → JSON.parse（失败抛 PermiaError(STORAGE_ERROR)，提示文件路径）
  → 文件不存在 → 使用空结构（自动创建）
  → 遍历 userRoles 构建 _roleUsers 反向索引

close():
  → clearTimeout(_debounceTimer)
  → await _writeToDisk()（确保进程退出前数据落盘）
```

**写操作（所有写方法统一流程）**：

```
1. 更新 _data 内存副本（同步）
2. 如涉及 userRoles 变更，同步更新 _roleUsers 反向索引（diff 算法）
3. _scheduleWrite()：debounce 100ms 触发 _writeToDisk()（含并发写入保护）
```

**并发写入保护（🆕 v2.4）**：

```typescript
private _writing = false;        // 当前是否正在执行 fs.writeFile
private _pendingWrite = false;   // 写入期间是否有新的写请求待处理

private _scheduleWrite(): void {
  if (this._debounceTimer) clearTimeout(this._debounceTimer);
  this._debounceTimer = setTimeout(() => this._writeToDisk(), 100);
}

private async _writeToDisk(): Promise<void> {
  // 🔴 并发锁：如果正在写入，标记 pending 等下一轮
  if (this._writing) {
    this._pendingWrite = true;
    return;
  }
  this._writing = true;
  try {
    const json = JSON.stringify(this._data, null, 2);
    await fs.writeFile(this.filePath, json, 'utf-8');
  } catch (e) {
    throw new PermiaError(
      PermiaErrorCode.STORAGE_ERROR,
      `Failed to write ${this.filePath}: ${(e as Error).message}`,
    );
  } finally {
    this._writing = false;
    // 写入完成后检查是否有待处理的写请求 → 重新调度
    if (this._pendingWrite) {
      this._pendingWrite = false;
      this._scheduleWrite();
    }
  }
}
```

> 💡 **为什么需要并发锁**：debounce 100ms 保证了高频写操作的合并，但当 `_writeToDisk` 正在执行异步 `fs.writeFile` 时（耗时取决于文件大小），新的 debounce 定时器可能再次触发 `_writeToDisk`，导致两次 `writeFile` 并发执行、文件内容交错损坏。`_writing` 锁确保同一时刻只有一次磁盘写入。

**setUserRoles 反向索引 diff 算法**：

```typescript
private _updateRoleUsersIndex(userId: string, newRoleIds: string[]): void {
  const oldIds = this._data.userRoles[userId] ?? [];
  const removed = oldIds.filter(r => !newRoleIds.includes(r));
  const added   = newRoleIds.filter(r => !oldIds.includes(r));

  for (const r of removed) {
    this._roleUsers.get(r)?.delete(userId);
  }
  for (const r of added) {
    if (!this._roleUsers.has(r)) this._roleUsers.set(r, new Set());
    this._roleUsers.get(r)!.add(userId);
  }
}
```

**并发安全**：

```yaml
Node.js 单线程事件循环，内存操作无竞态
磁盘写入通过 _writing 锁保证同一时刻只有一次 writeFile（🆕 v2.4）
多进程部署 → 各进程独立内存副本，数据不一致 → 改用 MonSQLizeAdapter
```

## 三、缓存层（src/cache/permission-cache.ts）

```typescript
import type { PermissionRule } from '../types';

interface CacheEntry {
  rules:     PermissionRule[];
  expiredAt: number;              // performance.now() + ttl（单调时钟）
}

interface CacheConfig {
  enabled: boolean;
  ttl:     number;                // 默认 300000 ms（5 分钟）
}

class PermissionCache {
  private enabled: boolean;
  private ttl: number;
  private store: Map<string, CacheEntry> = new Map();

  constructor(config: CacheConfig = { enabled: true, ttl: 300000 }) {
    this.enabled = config.enabled;
    this.ttl = config.ttl;
  }

  get(userId: string): PermissionRule[] | null {
    if (!this.enabled) return null;
    const entry = this.store.get(userId);
    if (!entry) return null;
    if (performance.now() > entry.expiredAt) {
      this.store.delete(userId);
      return null;
    }
    return entry.rules;
  }

  set(userId: string, rules: PermissionRule[]): void {
    if (!this.enabled) return;
    this.store.set(userId, {
      rules,
      expiredAt: performance.now() + this.ttl,
    });
  }

  invalidate(userId: string): void {
    this.store.delete(userId);
  }

  invalidateAll(): void {
    this.store.clear();
  }
}
```

> 💡 **过期判断**：使用 `performance.now()`（单调时钟）而非 `Date.now()`，避免系统时钟跳变（NTP 同步、VM 快照恢复等）导致缓存提前过期或永不过期。


> 💡 **`write` 语义**：`write` 是 `create + update` 的快捷方式。
> - **规则侧**（`matchAction`）：`matchAction('write', 'create')` → `true`，`matchAction('write', 'delete')` → `false`。规则存储层仍存储原始 action 值，展开仅在匹配时发生。
> - **请求侧**（`checker.can`）：`can(userId, 'write', resource)` 展开为 `can('create') && can('update')`（AND 语义）。

### 5.2 validation.ts

```typescript
import { DB_ACTIONS, API_ACTIONS, HTTP_METHODS } from './constants';
import { PermiaError, PermiaErrorCode } from '../core/errors';

/**
 * 校验资源路径格式
 */
export function validateResource(resource: string): void {
  // 合法格式:
  //   'db:<name>'  /  'db:<name>:<field>'  /  'db:*'  /  'db:<name>:*'
  //   '<METHOD>:<path>'  — METHOD 在 HTTP_METHODS 中，path 以 / 开头
  //   '*'
  if (resource === '*') return;

  if (resource.startsWith('db:')) {
    const parts = resource.split(':');
    if (parts.length < 2 || parts.length > 3) {
      throw new PermiaError(PermiaErrorCode.INVALID_RESOURCE_PATH, `Invalid db resource: '${resource}'`);
    }
    return;
  }

  const colonIdx = resource.indexOf(':');
  if (colonIdx === -1) {
    throw new PermiaError(PermiaErrorCode.INVALID_RESOURCE_PATH, `Invalid resource format: '${resource}'`);
  }

  const method = resource.slice(0, colonIdx);
  const path = resource.slice(colonIdx + 1);

  if (!HTTP_METHODS.includes(method as any)) {
    throw new PermiaError(PermiaErrorCode.INVALID_RESOURCE_PATH, `Invalid HTTP method: '${method}'`);
  }
  if (!path.startsWith('/') && path !== '*') {
    throw new PermiaError(PermiaErrorCode.INVALID_RESOURCE_PATH, `Path must start with '/' or be '*': '${path}'`);
  }
}

/**
 * 校验 action 是否与资源类型匹配
 */
export function validateAction(action: string, resource: string): void {
  if (action === '*') return;

  if (resource.startsWith('db:')) {
    if (!DB_ACTIONS.includes(action as any)) {
      throw new PermiaError(PermiaErrorCode.INVALID_ACTION, `Invalid action '${action}' for db resource. Allowed: ${DB_ACTIONS.join(', ')}`);
    }
    return;
  }

  if (resource === '*') return;   // 全局通配资源，action 任意

  // 接口资源
  if (!API_ACTIONS.includes(action as any)) {
    throw new PermiaError(PermiaErrorCode.INVALID_ACTION, `Invalid action '${action}' for API resource. Allowed: ${API_ACTIONS.join(', ')}`);
  }
}

/**
 * 校验初始化状态
 */
export function validateInitialized(initialized: boolean): void {
  if (!initialized) {
    throw new PermiaError(
      PermiaErrorCode.NOT_INITIALIZED,
      'Must call await permia.init() before using any API'
    );
  }
}
```

```typescript
import type { PermissionRule } from '../types';

/** 去除 type+action+resource 三者完全相同的重复规则 */
export function deduplicate(rules: PermissionRule[]): PermissionRule[] {
  const seen = new Set<string>();
  return rules.filter(rule => {
    const key = `${rule.type}|${rule.action}|${rule.resource}`;
    if (seen.has(key)) return false;
    seen.add(key);
    return true;
  });
}

/** 按类型拆分规则 */
export function splitByType(rules: PermissionRule[]): {
  allows: PermissionRule[];
  denies: PermissionRule[];
} {
  return {
    allows: rules.filter(r => r.type === 'allow'),
    denies: rules.filter(r => r.type === 'deny'),
  };
}
```
