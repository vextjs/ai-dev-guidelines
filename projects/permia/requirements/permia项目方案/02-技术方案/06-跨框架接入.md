## 一、接入模式总览

| 模式 | 时机 | 核心 API | 适用场景 |
|------|------|---------|---------|
| **实时鉴权** | 每次 HTTP 请求 | `assert(userId, 'invoke', resource)` | 中间件拦截，每请求验证 |
| **权限拉取** | 用户登录时 | `getResources(userId, 'invoke')` | 写入 JWT Token / 前端初始化菜单 |
| **字段过滤** | 查询数据后 | `filterFields(userId, 'db:xx', data)` | 敏感字段脱敏返回 |
| **前端判断** | 页面渲染时 | `matchResource(pattern, path)` | 按钮/栏目显隐控制 |

**接入流程**：
```
其他框架 userId → permia.users.assign(userId, roleId)
用户登录 → getResources → 写 JWT
每次请求 → assert(userId, 'invoke', METHOD:path) → 通过/403
查询后   → filterFields(userId, 'db:collection', data) → 脱敏数据
```

## 二、Express 完整接入示例

```javascript
const express = require('express');
const jwt = require('jsonwebtoken');
const { Permia, PermiaError, matchResource } = require('permia');

const app = express();
const SECRET = process.env.JWT_SECRET;

// ① 初始化 permia
const permia = new Permia();
await permia.init();

// ② 预置权限（通常由管理后台动态配置，此处为示例）
await permia.roles.create('admin', { label: '管理员' });
await permia.roles.allow('admin', 'invoke', '*:/api/*');
await permia.users.assign('user-001', 'admin');

// ③ 鉴权中间件（每次请求执行）
app.use(async (req, res, next) => {
  const resource = `${req.method}:${req.path}`;  // 'DELETE:/api/users/123'
  try {
    await permia.assert(req.userId, 'invoke', resource);
    next();
  } catch (e) {
    if (e.code === PermiaError.PERMISSION_DENIED)
      return res.status(403).json({ code: e.code, message: e.message });
    next(e);   // STORAGE_ERROR 等透传给错误中间件
  }
});

// ④ 登录接口 — 拉取权限写入 Token
app.post('/api/auth/login', async (req, res) => {
  const userId = await authenticate(req.body);   // 业务层认证
  const resources = await permia.getResources(userId, 'invoke');
  // resources = ['*:/api/users/*', 'GET:/api/articles', ...]
  const token = jwt.sign({ userId, resources }, SECRET, { expiresIn: '8h' });
  res.json({ token });
  // 前端拿到 resources 渲染菜单和按钮
});

// ⑤ 查询后字段过滤
app.get('/api/users/:id', async (req, res) => {
  const user = await db.findUser(req.params.id);
  const safe = await permia.filterFields(req.userId, 'db:users', user);
  res.json(safe);   // 仅含当前用户有权查看的字段
});

// ⑥ 优雅关闭
process.on('SIGTERM', async () => {
  await permia.close();
  process.exit(0);
});

app.listen(3000);
```

## 三、Koa 接入示例

```javascript
const Koa = require('koa');
const { Permia, PermiaError } = require('permia');

const app = new Koa();
const permia = new Permia();
await permia.init();

// 鉴权中间件
app.use(async (ctx, next) => {
  const resource = `${ctx.method}:${ctx.path}`;  // Koa 用 ctx.method / ctx.path
  try {
    await permia.assert(ctx.state.userId, 'invoke', resource);
    await next();
  } catch (e) {
    if (e.code === PermiaError.PERMISSION_DENIED) {
      ctx.status = 403;
      ctx.body = { code: e.code, message: e.message };
    } else {
      throw e;
    }
  }
});
```

## 四、vext 框架接入示例

```javascript
const { Permia, PermiaError } = require('permia');
const permia = new Permia();
await permia.init();

// 注册 vext 鉴权中间件
app.use('auth:permission', async (ctx, next) => {
  const resource = `${ctx.method}:${ctx.path}`;
  try {
    await permia.assert(ctx.userId, 'invoke', resource);
    await next();
  } catch (e) {
    if (e.code === PermiaError.PERMISSION_DENIED)
      ctx.throw(403, e.message);
    throw e;
  }
});
```

## 五、前端菜单 / 按钮权限判断

登录时服务端返回 `resources`（通过 `getResources` 获取），前端存入 Token 或状态管理，渲染时使用 `matchResource` 判断。

```javascript
// 服务端（登录接口）
const resources = await permia.getResources(userId, 'invoke');
// → ['*:/api/users/*', 'GET:/api/articles', 'POST:/api/articles']
// 写入 JWT payload 或直接返回给前端

// ---

// 前端（React / Vue 等）
import { matchResource } from 'permia';  // 需要打包时引入，或单独提取此函数

const resources = useAuth().resources;   // 从状态管理取出

// 5.1 按钮级判断
const canDeleteUser   = resources.some(p => matchResource(p, 'DELETE:/api/users/123'));
// → true（*:/api/users/* 匹配）

const canCreateArticle = resources.some(p => matchResource(p, 'POST:/api/articles'));
// → true

const canDeleteArticle = resources.some(p => matchResource(p, 'DELETE:/api/articles/1'));
// → false（无匹配规则）

// 5.2 菜单栏目级判断（整个栏目是否可见）
const menus = [
  { name: '用户管理', testPath: 'GET:/api/users' },
  { name: '文章管理', testPath: 'GET:/api/articles' },
  { name: '系统设置', testPath: 'GET:/api/settings' },
];
const visibleMenus = menus.filter(m =>
  resources.some(p => matchResource(p, m.testPath))
);
// → [{ name: '用户管理', ... }, { name: '文章管理', ... }]
// 系统设置无权限不显示

// 5.3 路由守卫（拦截无权限页面）
router.beforeEach((to) => {
  const requiredResource = to.meta?.requiredResource;
  if (!requiredResource) return true;   // 无限制的路由直接放行
  const allowed = resources.some(p => matchResource(p, requiredResource));
  if (!allowed) return { path: '/403' };
});
```

## 六、跨服务接入（微服务场景）

```
场景：服务A 有自己的用户体系，通过 permia 管理权限

架构:
  服务A 的 userId（如 MongoDB ObjectId）→ 直接作为 permia 的 userId 使用
  permia 不关心 userId 的格式，任何 string 均可

接入步骤:
  1. 用户在服务A 注册后，调用 permia.users.assign(userId, defaultRole)
  2. 用户登录时，调用 permia.getResources(userId, 'invoke') 获取权限写入 Token
  3. 每次 API 请求，通过鉴权中间件调用 permia.assert(userId, 'invoke', resource)
  4. 权限变更时，调用 permia.roles.allow/deny 修改角色规则，缓存自动失效
```
